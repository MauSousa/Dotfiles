# Function to create and enter a directory
function crear-entrar(){
    mkdir -p $1 
    cd $1
}

# Function for MySQL
function entrarSql() {
	mysql -u $1 -p
}

# Function to delete files
function rmk() {
	scrub -p dod $1
	shred -zun 10 -v $1
}

# Function to copy and push dotfiles to repository
function cdf() {
 cp ~/.zshrc ~/.zshenv ~/.config/dotfiles/zsh; cp ~/.config/starship/starship.toml ~/.config/dotfiles/starship/; cp -r ~/.config/rofi/ ~/.config/dotfiles/
}

# Function to copy bspwm config, polybar, sxhkdrc
function cbc {
	cp  ~/.config/bspwm/bspwmrc ~/.config/dotfiles/bspwm; cp ~/.config/sxhkd/sxhkdrc ~/.config/dotfiles/sxhkd;	cp ~/.config/polybar/config.ini ~/.config/polybar/launch.sh ~/.config/dotfiles/polybar; cp ~/.config/bin/kernel_info.sh ~/.config/bin/i3_weather.sh ~/.config/bin/uptime.sh ~/.config/dotfiles/bin
}

# Function to copy wallpapers
function cpwall() {
  cp -r ~/Pictures ~/.config/dotfiles
}

# Fuction to copy lua nvim config
function cpluanvim() {
  cp ~/.config/nvim/init.lua ~/.config/dotfiles/lua-neovim; cp ~/.config/nvim/lua/user/*.lua ~/.config/dotfiles/lua-neovim/lua/user; cp ~/.config/nvim/lua/user/lsp/*.lua ~/.config/dotfiles/lua-neovim/lua/user/lsp; cp ~/.config/nvim/lua/user/lsp/settings/*.lua ~/.config/dotfiles/lua-neovim/lua/user/lsp/settings  
}

# Function to create and activate python virtual env 
function pyenv() {
	python3 -m venv $1
	source $1/bin/activate
}

# Function to kill Discord process
function kd() {
	killall Discord && killall Discord
}

# Function to run TS files with Deno
function dr() {
	deno run $1
}

# Function to bundle TS file
function dnb() {
	deno bundle $1 $2
}

# Function to remove directories
function rd() {
  rm -r $1
}
